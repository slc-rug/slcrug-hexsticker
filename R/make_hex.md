```r
library(viridis)
library(hexSticker)

ut <- utah %>%
    ggplot(aes(fill = estimate, color = estimate)) + 
    geom_sf() + 
    coord_sf(xlim = c(-107, -116),
             ylim = c(36, 46)) + 
    scale_fill_viridis() + 
    scale_color_viridis() + 
    theme_minimal() +
    theme(legend.position="none") +
    theme(axis.text.x=element_blank(),
          axis.text.y=element_blank()) +
    theme_transparent()

ut
```
